- Initialize pyairtable: `api = Api(os.getenv("AIRTABLE_API_KEY"))`

**Step 2: Create `src/run_pipeline.py`** as the pipeline orchestrator per D-13, D-16.

```python
import sys
import os
sys.path.insert(0, os.path.dirname(__file__))

from pipeline.discover import discover_prospects
from pipeline.enrich import enrich_prospects
from pipeline.score import score_and_filter
from pipeline.upload import upload_to_airtable


def run():
    print("=" * 60)
    print("PROSPECT PIPELINE")
    print("=" * 60)

    print("\n[Stage 1/4] Discovery")
    print("-" * 40)
    raw = discover_prospects()
    print(f"  Result: {len(raw)} candidates discovered\n")

    print("[Stage 2/4] Enrichment")
    print("-" * 40)
    enriched = enrich_prospects(raw)
    enriched_ok = sum(1 for p in enriched if p.get("enrichment_status") == "enriched")
    enriched_fail = sum(1 for p in enriched if p.get("enrichment_status") == "enrichment-failed")
    print(f"  Result: {enriched_ok} enriched, {enriched_fail} failed\n")

    print("[Stage 3/4] Scoring")
    print("-" * 40)
    scored = score_and_filter(enriched, skip_list_path=os.path.join(os.path.dirname(__file__), "skip_list.json"))
    call_ready = [p for p in scored if p["status"] == "call-ready"]
    below = [p for p in scored if p["status"] == "below-threshold"]
    disqualified = [p for p in scored if p["status"] == "disqualified"]
    skipped = [p for p in scored if p["status"] == "skipped"]
    print(f"  Result: {len(call_ready)} call-ready, {len(below)} below-threshold, {len(disqualified)} disqualified, {len(skipped)} skipped\n")

    print("[Stage 4/4] Upload to Airtable")
    print("-" * 40)
    upload_to_airtable(scored)

    print("\n" + "=" * 60)
    print("PIPELINE COMPLETE")
    print(f"  Total prospects: {len(scored)}")
    print(f"  Call-ready: {len(call_ready)}")
    print(f"  Airtable: populated")
    print("=" * 60)


if __name__ == "__main__":
    run()
```

- No `input()` calls (SCOR-04: fully automated, no human review step)
- Clean terminal output showing each stage with separators (D-16)
- Skip list path resolved relative to the script's directory, not cwd
- Exit with non-zero code if any stage raises an unhandled exception
